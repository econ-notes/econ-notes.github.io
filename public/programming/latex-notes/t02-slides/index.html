%!! p-3-2 table of conetents are presented before new subsections 
\newif\ifoutlineshown
\outlineshownfalse

\AtBeginSection[]{
    \outlineshowntrue
    \begin{frame}
    \frametitle{Outline}
    \tableofcontents[currentsection]
    \end{frame}
}

\AtBeginSubsection[]{
    \ifoutlineshown
    \outlineshownfalse
    \else
    \begin{frame}
        \frametitle{Outline}
        \tableofcontents[currentsection,currentsubsection]
    \end{frame}
    \fi
}
